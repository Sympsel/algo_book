int cmp(const string& s1, const string& s2) {
    int sz1(s1.size()), sz2(s2.size());
    if (sz1 == sz2) {
        for (int i{}; i < sz1; ++i) {
            if (s1[i] > s2[i]) return 1;
            else if (s1[i] < s2[i]) return -1;
            else{}
        }
        return 0;
    }
    return sz1 > sz2;
}
